OBJECTIVE
A program that finds the sum of main diagonal and off diagonal elements of a m*n matrix.

METHODOLOGY
1. Create a matrix and define its elements.
2. Declare two variables which will store sum of main and opposite diagonal.
3. Now run a single for loop and extract main diagonals elements adding to the first variable, the elements with the same index are considered to be diagonal elements i.e. i=j
4. finding opposite diagonal elements i.e. (i,m-i-1) to the second variable.

INPUT
we need the size of array from ths user and the elements of array 

OUTPUT
sum of main diagonal elements and off diagonal elements will be displayed.

CODE
#include <iostream>
using namespace std;
    int main ()
    {    int array[10][10];
         int i, j, m, n, a = 0, sum = 0;
 
        printf("Enetr the order of the matix \n");
        scanf("%d %d", &m, &n);
 
        if (m == n ) 
        {
 
            printf("Enter the co-efficients of the matrix\n");
            for (i = 0; i < m; ++i)
            {
                for (j = 0; j < n; ++j)
                {
                    scanf("%d", &array[i][j]);
                }
            }
 
            printf("The given matrix is \n");
            for (i = 0; i < m; ++i) 
            {
                for (j = 0; j < n; ++j)
                {
                    printf(" %d", array[i][j]);
                }
                printf("\n");
            }
 
            for (i = 0; i < m; ++i) 
            {
                sum = sum + array[i][i];
                a = a + array[i][m - i - 1];
            }
 
            printf("\nThe sum of the main diagonal elements is = %d\n", sum);
            printf("The sum of the off diagonal elements is   = %d\n", a);
 
        }
 
        else
            printf("The given order is not square matrix\n");
        return 0;
    }
